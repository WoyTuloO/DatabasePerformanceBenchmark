CREATE (6)
C1) Dodanie zamówienia (nagłówek koszyka)

INSERT INTO orders (customer_id, shipping_address_id, status, total_cents, currency)
VALUES (:customer_id, :shipping_address_id, 'NEW', 0, :currency)
RETURNING order_id, created_at;


C2) Dodanie itemu do koszyka (auto line_no)

WITH next_line AS (
  SELECT COALESCE(MAX(line_no), 0) + 1 AS line_no
  FROM order_items
  WHERE order_id = :order_id
)
INSERT INTO order_items (order_id, line_no, product_id, quantity, unit_price_cents)
SELECT :order_id, nl.line_no, :product_id, :quantity, :unit_price_cents
FROM next_line nl
RETURNING order_id, line_no;


C3) Dodanie klienta

INSERT INTO customers (email, password_hash, first_name, last_name, phone)
VALUES (:email, :password_hash, :first_name, :last_name, :phone)
RETURNING customer_id;


C4) Dodanie adresu klienta

INSERT INTO customer_addresses (
  customer_id, address_label, country, city, postal_code, street, building_no, apartment_no
)
VALUES (
  :customer_id, :address_label, :country, :city, :postal_code, :street, :building_no, :apartment_no
)
RETURNING address_id;


C5) Dodanie produktu

INSERT INTO products (stock_keeping_unit, name, description, brand_id, category_id, base_price_cents, currency, active)
VALUES (:stock_keeping_unit, :name, :description, :brand_id, :category_id, :base_price_cents, :currency, TRUE)
RETURNING product_id;


C6) Upsert stanu magazynowego (create lub update)

INSERT INTO inventory (warehouse_id, product_id, quantity)
VALUES (:warehouse_id, :product_id, :quantity)
ON CONFLICT (warehouse_id, product_id)
DO UPDATE SET quantity = EXCLUDED.quantity,
              updated_at = now()
RETURNING warehouse_id, product_id, quantity;



READ (6)

R1) Pobranie zamówień klienta (to, o co prosiłeś)

SELECT o.order_id, o.status, o.total_cents, o.currency, o.created_at
FROM orders o
WHERE o.customer_id = :customer_id
ORDER BY o.created_at DESC
LIMIT :limit OFFSET :offset;


R2) Pobranie koszyka klienta (ostatnie NEW)

SELECT o.order_id, o.created_at
FROM orders o
WHERE o.customer_id = :customer_id
  AND o.status = 'NEW'
ORDER BY o.created_at DESC
LIMIT 1;


R3) Zawartość koszyka (pozycje + produkty)

SELECT
  oi.line_no,
  oi.product_id,
  p.stock_keeping_unit,
  p.name,
  oi.quantity,
  oi.unit_price_cents,
  (oi.quantity * oi.unit_price_cents) AS line_total_cents
FROM order_items oi
JOIN products p ON p.product_id = oi.product_id
WHERE oi.order_id = :order_id
ORDER BY oi.line_no;


R4) Sprawdzenie dostępności produktu (suma stanów w magazynach)

SELECT
  p.product_id,
  p.stock_keeping_unit,
  p.name,
  p.active,
  COALESCE(SUM(i.quantity), 0) AS total_stock
FROM products p
LEFT JOIN inventory i ON i.product_id = p.product_id
WHERE p.product_id = :product_id
GROUP BY p.product_id;


R5) Lista produktów (filtry + prosta wyszukiwarka)

SELECT p.product_id, p.stock_keeping_unit, p.name, p.base_price_cents, p.currency, p.active
FROM products p
WHERE (:active IS NULL OR p.active = :active)
  AND (:q IS NULL OR p.name ILIKE '%' || :q || '%' OR p.stock_keeping_unit ILIKE '%' || :q || '%')
ORDER BY p.created_at DESC
LIMIT :limit OFFSET :offset;


R6) Pobranie szczegółów zamówień klienta z produktami, marką, kategorią i płatnością

SELECT
  o.order_id, o.created_at, o.status, o.total_cents, o.currency,
  pmt.method, pmt.status AS payment_status, pmt.paid_at,
  oi.line_no, oi.quantity, oi.unit_price_cents,
  pr.stock_keeping_unit, pr.name AS product_name,
  b.name AS brand_name,
  c.name AS category_name
FROM orders o
JOIN order_items oi    ON oi.order_id = o.order_id
JOIN products pr       ON pr.product_id = oi.product_id
LEFT JOIN brands b     ON b.brand_id = pr.brand_id
LEFT JOIN categories c ON c.category_id = pr.category_id
LEFT JOIN payments pmt ON pmt.order_id = o.order_id
WHERE o.customer_id = :customer_id
ORDER BY o.created_at DESC, oi.line_no;


UPDATE (6)


U1) Zmiana ilości przedmiotów w koszyku

UPDATE order_items
SET quantity = :quantity
WHERE order_id = :order_id
  AND line_no = :line_no
RETURNING order_id, line_no, quantity;


U2) Zmiana statusu zamówienia (np. NEW → PAID/SHIPPED)

UPDATE orders
SET status = :status
WHERE order_id = :order_id
RETURNING order_id, status;


U3) Zmiana statusu/dostępności produktu (active TRUE/FALSE)

UPDATE products
SET active = :active
WHERE product_id = :product_id
RETURNING product_id, active;


U4) Przeliczenie total zamówienia na podstawie itemów (ważne po zmianach koszyka)

UPDATE orders o
SET total_cents = COALESCE((
  SELECT SUM(quantity * unit_price_cents)
  FROM order_items oi
  WHERE oi.order_id = o.order_id
), 0)
WHERE o.order_id = :order_id
RETURNING order_id, total_cents;


U5) Aktualizacja stanu w magazynie (ustawienie konkretnej ilości)

UPDATE inventory
SET quantity = :quantity,
    updated_at = now()
WHERE warehouse_id = :warehouse_id
  AND product_id = :product_id
RETURNING warehouse_id, product_id, quantity;


U6) Hurtowa zmiana cen w kategorii (bardziej „zaawansowane” UPDATE)

UPDATE products
SET base_price_cents = GREATEST(0, (base_price_cents * :multiplier)::int)
WHERE category_id = :category_id
RETURNING product_id, base_price_cents;


DELETE (6)

D1) Usunięcie itemu z koszyka (linia)

DELETE FROM order_items
WHERE order_id = :order_id
  AND line_no = :line_no
RETURNING order_id, line_no;


D2) Usunięcie całego koszyka (zamówienia) — kaskadowo usunie itemy

DELETE FROM orders
WHERE order_id = :order_id
  AND status = 'NEW'
RETURNING order_id;


D3) Usunięcie adresu klienta

DELETE FROM customer_addresses
WHERE address_id = :address_id
RETURNING address_id;


D4) Usunięcie klienta (kaskadowo adresy; zamówienia wg Twoich FK zostaną zablokowane jeśli istnieją)

DELETE FROM customers
WHERE customer_id = :customer_id
RETURNING customer_id;


D5) Usunięcie magazynu (kaskadowo inventory)

DELETE FROM warehouses
WHERE warehouse_id = :warehouse_id
RETURNING warehouse_id;


D6) „Twarde” usunięcie produktu (uwaga: nie usunie się, jeśli istnieją order_items na ten produkt)

DELETE FROM products
WHERE product_id = :product_id
RETURNING product_id;
